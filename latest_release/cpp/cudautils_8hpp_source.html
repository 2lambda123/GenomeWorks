70 inline void gpu_assert(cudaError_t code, const char* file, int line)
71 {
72 #ifdef GW_DEVICE_SYNCHRONIZE
73  // This device synchronize forces the most recent CUDA call to fully
74  // complete, increasing the chance of catching the CUDA error near the
75  // offending function. Only run if existing code is success to avoid
76  // potentially overwriting previous error code.
77  if (code == cudaSuccess)
78  {
79  code = cudaDeviceSynchronize();
80  }
81 #endif
82 
83  if (code != cudaSuccess)
84  {
85  std::string err = "GPU Error:: " +
86  std::string(cudaGetErrorString(code)) +
87  " " + std::string(file) +
88  " " + std::to_string(line);
89  GW_LOG_ERROR("{}\n", err);
90  // In Debug mode, this assert will cause a debugger trap
91  // which is beneficial when debugging errors.
92  assert(false);
93  std::abort();
94  }
95 }
