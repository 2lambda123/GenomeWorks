168  {

171  if (pushed_last_element_)

186  cga_optional_t<T> get_next_element()
187  {
188  std::unique_lock<std::mutex> ul(mutex_);
189 
190  while (data_.empty() && !pushed_last_element_)
191  {
192  condition_variable_.wait(ul);
193  }
194 
195  const bool no_elements_left = pushed_last_element_ && data_.empty();
196  if (no_elements_left)

199  }

203  data_.pop_back();
