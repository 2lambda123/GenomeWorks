60 inline void gpu_assert(cudaError_t code, const char* file, int line)
61 {
62 #ifdef CGA_DEVICE_SYNCHRONIZE
63  // This device synchronize forces the most recent CUDA call to fully
64  // complete, increasing the chance of catching the CUDA error near the
65  // offending function. Only run if existing code is success to avoid
66  // potentially overwriting previous error code.
67  if (code == cudaSuccess)
68  {
69  code = cudaDeviceSynchronize();
70  }
71 #endif
72 
73  if (code != cudaSuccess)
74  {
75  std::string err = "GPU Error:: " +
76  std::string(cudaGetErrorString(code)) +
77  " " + std::string(file) +
78  " " + std::to_string(line);
79  CGA_LOG_ERROR("{}\n", err);
80  // In Debug mode, this assert will cause a debugger trap
81  // which is beneficial when debugging errors.
82  assert(false);
83  std::abort();
84  }
85 }
