33 inline std::string generate_random_genome(const int32_t length, std::minstd_rand& rng)
34 {
35  const char alphabet[4] = {'A', 'C', 'G', 'T'};
36  std::uniform_int_distribution<int32_t> random_index(0, 3);
37  std::string genome = "";
38  for (int32_t i = 0; i < length; i++)
39  {
40  genome += alphabet[random_index(rng)];
41  }
42  return genome;
43 }

45 inline std::string generate_random_sequence(const std::string& backbone, std::minstd_rand& rng, int max_mutations, int max_insertions, int max_deletions, std::vector<std::pair<int, int>>* ranges = nullptr)
46 {
47  throw_on_negative(max_mutations, "max_mutations cannot be negative.");
48  throw_on_negative(max_insertions, "max_insertions cannot be negative.");
49  throw_on_negative(max_deletions, "max_deletions cannot be negative.");
50 
51  const char alphabet[4] = {'A', 'C', 'G', 'T'};
52  std::uniform_int_distribution<int> random_base(0, 3);
53 
54  std::string sequence = backbone;
55  std::vector<std::pair<int, int>> full_range(1, std::make_pair(0, get_size<int>(backbone)));
56 
57  if (ranges == nullptr)
58  {
59  ranges = &full_range;
60  }
61 
62  // max_insertions, max_deletions and max_mutations are capped to range length
63  for (auto range : *ranges)
64  {
65  int start_index = range.first;
66  int end_index = range.second;
67 
68  throw_on_negative(start_index, "start_index of the range cannot be negative.");
69  throw_on_negative(end_index - start_index, "end_index of the range cannot be smaller than start_index.");
70 
71  if (get_size<int>(backbone) < end_index)
72  throw std::invalid_argument("end_index should be smaller than backbone's length.");
73 
74  int range_length = end_index - start_index;
75  std::string substring = backbone.substr(start_index, range_length);
76 
77  std::uniform_real_distribution<double> random_prob(0, 1);
78 
79  for (int j = 0; j < std::min(max_deletions, range_length); j++)
80  {
81  if (random_prob(rng) > 0.5)
82  {
83  int length = substring.length();
84  std::uniform_int_distribution<int> random_del_pos(0, length - 1);
85  int del_pos = random_del_pos(rng);
86  substring.erase(del_pos, 1);
87  }
88  }
89 
90  for (int j = 0; j < std::min(max_insertions, range_length); j++)
91  {
92  if (random_prob(rng) > 0.5)
93  {
94  int length = substring.length();
95  std::uniform_int_distribution<int> random_ins_pos(0, length);
96  int ins_pos = random_ins_pos(rng);
97  int ins_base = random_base(rng);
98 
99  substring.insert(ins_pos, 1, alphabet[ins_base]);
100  }
101  }
102 
103  int length = substring.length();
104  if (length > 0)
105  {
106  std::uniform_int_distribution<int> random_mut_pos(0, length - 1);
107  for (int j = 0; j < std::min(max_mutations, range_length); j++)
108  {
109  if (random_prob(rng) > 0.5)
110  {
111  int mut_pos = random_mut_pos(rng);
112  int swap_base = random_base(rng);
113  substring[mut_pos] = alphabet[swap_base];
114  }
115  }
116  }
117 
118  if (start_index < static_cast<int>(sequence.length()))
119  {
120  sequence.replace(start_index, range_length, substring);
121  }
122  }
123 
124  return sequence;
125 }

127 inline std::vector<std::string> generate_random_sequences(std::string const& backbone,
128  int n,
129  std::minstd_rand& rng,
130  int max_mutations = 1,
131  int max_insertion = 1,
132  int max_deletions = 1)
133 {
134  throw_on_negative(n, "n cannot be negative!");
135  std::vector<std::string> sequences;
136  sequences.reserve(n);
137  sequences.push_back(backbone);
138  for (int i = 1; i < n; i++)
139  sequences.push_back(generate_random_sequence(backbone, rng, max_mutations, max_insertion, max_deletions));
140 
141  return sequences;
142 }

144 inline void reverse_complement(const char* src, const int32_t length, char* dest)
145 {
146  for (int32_t pos = 0; pos < length; pos++)
147  {
148  switch (char nucleotide = src[length - 1 - pos])
149  {
150  case 'A': dest[pos] = 'T'; break;
151  case 'T': dest[pos] = 'A'; break;
152  case 'C': dest[pos] = 'G'; break;
153  case 'G': dest[pos] = 'C'; break;
154  default: dest[pos] = nucleotide;
155  }
156  }
157 }
