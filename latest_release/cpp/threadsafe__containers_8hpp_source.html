138  void add_new_element(const T& element) // not using reference in order to support both rvalue and lvalue
139  {

176  void signal_pushed_last_element()
177  {
178  {
179  std::lock_guard<std::mutex> lg(mutex_);
180  if (pushed_last_element_)
181  {
182  throw std::logic_error("ThreadsafeProducerConsumer: called signal_pushed_last_element() more than once");
183  }
184  pushed_last_element_ = true;
185  }
186  condition_variable_.notify_all();

195  gw_optional_t<T> get_next_element()
196  {
197  std::unique_lock<std::mutex> ul(mutex_);
198 
199  while (data_.empty() && !pushed_last_element_)
200  {
201  condition_variable_.wait(ul);
202  }
203 
204  const bool no_elements_left = pushed_last_element_ && data_.empty();
205  if (no_elements_left)
206  {
207  return gw_nullopt;
208  }
209  else
210  {
211  gw_optional_t<T> res = std::move(data_.back());
212  data_.pop_back();
213  return res;
214  }
215  }
