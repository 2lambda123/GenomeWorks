36 struct PairHash
37 {
38 public:
40  template <class T1, class T2>
41  size_t operator()(const std::pair<T1, T2>& pair) const
42  {
43  size_t hash_1 = std::hash<T1>()(pair.first);
44  size_t hash_2 = std::hash<T2>()(pair.second);
45  return hash_1 ^ hash_2;
46  }
47 };

139  void update_adject_nodes(edge_t edge)
140  {
141  auto find_node = adjacent_nodes_.find(edge.first);
142  if (find_node == adjacent_nodes_.end())
143  {
144  adjacent_nodes_.insert({edge.first, {edge.second}});

208  {

252  void add_edge(node_id_t node_id_from, node_id_t node_id_to, edge_weight_t weight = 0)
253  {
254  auto edge = edge_t(node_id_from, node_id_to);
255  auto edge_reversed = edge_t(node_id_to, node_id_from);
256  if (!directed_edge_exists(edge) && !directed_edge_exists(edge_reversed))
257  {
258  edges_.insert({edge, weight});
259  update_adject_nodes(edge);

261  }
