74 inline void resize_windows(std::vector<std::vector<std::string>>& windows, const int32_t total_windows)
75 {
76  if (total_windows >= 0)
77  {
78  if (get_size(windows) > total_windows)
79  {
80  windows.erase(windows.begin() + total_windows, windows.end());
81  }
82  else if (get_size(windows) < total_windows)
83  {
84  int32_t windows_read = windows.size();
85  while (get_size(windows) != total_windows)
86  {
87  windows.push_back(windows[windows.size() - windows_read]);
88  }
89  }
90 
91  assert(windows.size() == total_windows);
92  }
93 }

114 inline void parse_cudapoa_file(std::vector<std::vector<std::string>>& windows, const std::string& filename, int32_t total_windows)
115 {
116  std::ifstream infile(filename);
117  if (!infile.good())
118  {
119  throw std::runtime_error("Cannot read file " + filename);
120  }
121  std::string line;
122  int32_t num_sequences = 0;
123  while (std::getline(infile, line))
124  {
125  if (num_sequences == 0)
126  {
127  std::istringstream iss(line);
128  iss >> num_sequences;
129  windows.emplace_back(std::vector<std::string>());
130  }
131  else
132  {
133  windows.back().push_back(line);
134  num_sequences--;
135  }
136  }
137 
138  resize_windows(windows, total_windows);
139 }

149 inline void parse_fasta_files(std::vector<std::vector<std::string>>& windows, const std::vector<std::string>& input_paths, const int32_t total_windows)
150 {
151  const int32_t min_sequence_length = 0;
152  const int32_t num_input_files = input_paths.size();
153  windows.resize(num_input_files);
154  for (int32_t i = 0; i < num_input_files; i++)
155  {
156  std::shared_ptr<io::FastaParser> fasta_parser = io::create_kseq_fasta_parser(input_paths[i], min_sequence_length, false);
157  int32_t num_reads = fasta_parser->get_num_seqences();
158  for (int32_t idx = 0; idx < num_reads; idx++)
159  {
160  windows[i].push_back(fasta_parser->get_sequence_by_id(idx).seq);
161  }
162  }
163  resize_windows(windows, total_windows);
164 }
