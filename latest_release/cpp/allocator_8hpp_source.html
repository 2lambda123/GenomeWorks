116  pointer allocate(std::size_t n,
117  const std::vector<cudaStream_t>& streams = {{0}})
118  {
119  static_cast<void>(streams);
120  void* ptr = nullptr;
121  cudaError_t err = cudaMalloc(&ptr, n * sizeof(T));
122  if (err == cudaErrorMemoryAllocation)
123  {
124  // Clear the error from the runtime...
125  err = cudaGetLastError();
126  // Did a different (async) error happen in the meantime?
127  if (err != cudaErrorMemoryAllocation)
128  {
129  GW_CU_CHECK_ERR(err);
130  }
131  throw device_memory_allocation_exception();
132  }
133  GW_CU_CHECK_ERR(err);
134  return static_cast<pointer>(ptr);
135  }

250  pointer allocate(std::size_t n,
251  const std::vector<cudaStream_t>& streams = {{0}})
252  {
253  assert(!streams.empty());
254 
255  if (!memory_resource_)
256  {
257  GW_LOG_ERROR("{}\n", "ERROR:: Trying to allocate memory from an default-constructed CachingDeviceAllocator. Please assign a non-default-constructed CachingDeviceAllocator before performing any memory operations.");
258  assert(false);
259  std::abort();
260  }
261  void* ptr = nullptr;
262  cudaError_t err = memory_resource_->DeviceAllocate(&ptr, n * sizeof(T), streams);
263  if (err == cudaErrorMemoryAllocation)
264  {
265  throw device_memory_allocation_exception();
266  }
267  GW_CU_CHECK_ERR(err);
268  return static_cast<pointer>(ptr);
269  }

280  assert(false);

282  }

284  GW_CU_ABORT_ON_ERR(memory_resource_->DeviceFree(p));
